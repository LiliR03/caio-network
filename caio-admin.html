How to set up Google Sheets
  1. Open Google Sheets and create a new spreadsheet named CAIO Network.
  2. Go to Extensions → Apps Script. Delete any code there.
  3. Paste the script below and click Save (disk icon).
  4. Click Deploy → New deployment. Type: Web app. Execute as: Me. Access: Anyone. Click Deploy & authorize.
  5. Copy the Web App URL and paste it above, then click Connect.
  6. Share this URL with classmates — they paste it into their own copy to connect.
const SHEET_NAME = 'Contacts';

function doGet(e) {
  const action = e.parameter.action;
  if (action === 'getAll') return getAll();
  return ContentService.createTextOutput('ok');
}

function doPost(e) {
  const data = JSON.parse(e.postData.contents);
  if (data.action === 'add')    return addRow(data.contact);
  if (data.action === 'delete') return deleteRow(data.id);
  return res({ok: true});
}

function getAll() {
  const sheet = getSheet();
  const rows  = sheet.getDataRange().getValues();
  if (rows.length <= 1) return res([]);
  const [headers, ...data] = rows;
  const contacts = data.map(r =>
    Object.fromEntries(headers.map((h, i) => [h, r[i]]))
  );
  return res(contacts);
}

function addRow(c) {
  const sheet   = getSheet();
  const headers = sheet.getRange(1,1,1,sheet.getLastColumn()).getValues()[0];
  if (!headers[0]) {
    sheet.getRange(1,1,1,7).setValues(
      [['id','first','last','industry','role','company','email']]
    );
  }
  sheet.appendRow([c.id,c.first,c.last,c.industry,c.role||'',c.company||'',c.email||'']);
  return res({ok: true});
}

function deleteRow(id) {
  const sheet = getSheet();
  const data  = sheet.getDataRange().getValues();
  for (let i = data.length - 1; i >= 1; i--) {
    if (String(data[i][0]) === String(id)) { sheet.deleteRow(i+1); break; }
  }
  return res({ok: true});
}

function getSheet() {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  return ss.getSheetByName(SHEET_NAME) || ss.insertSheet(SHEET_NAME);
}

function res(data) {
  return ContentService
    .createTextOutput(JSON.stringify(data))
    .setMimeType(ContentService.MimeType.JSON);
}
